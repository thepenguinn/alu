\section{Testing the Sixteen Bit ALU}

\subsection {Ring Oscillator to Generate a Clock Signal}

\begin{verbatim}
    /*
     * Yepp, everything is structural ! ;)
     * */
    module ring_oscillator(input logic en,
        output logic out);

        wire s4, s2, s3;
        wire noten, nots3;

        nor (noten, en, en);
        nor (nots3, s3, s3);
        nor (s4, noten, nots3);

        nor #300 (s2, out, out);
        nor #300 (s3, s2, s2);
        nor #300 (out, s4, s4);

    endmodule
\end{verbatim}


\subsection {Testbench}

\begin{verbatim}
    module testbench;

        reg en;
        wire clkout;

        reg [15:0] aluina, aluinb;
        reg [2:0] aluop;
        reg aluon;
        wire [3:0] alucount;
        wire [16:0] aluout;

        integer i, j;

        ring_oscillator rosci (
            .en(en),
            .out(clkout)
        );

        alu16 alu (
            .clk(clkout),
            .on(aluon),
            .ina(aluina),
            .inb(aluinb),
            .op(aluop),
            .out(aluout),
            .count(alucount)
        );

        initial begin

            en = 1'b0;
            #900;
            en = 1'b1;

            aluina = 16'b0111_0000_0000_0011;
            aluinb = 16'b1100_0000_0000_0011;

            aluop = 3'b011;

            aluon = 1'b0;
            #100;
            aluon = 1'b1;
            #100;
            aluon = 1'b0;

            j = 0;
            i = 0;

        end

        initial begin
            $display("  Count Aluout                            Time ");
        end

        always @(posedge clkout) begin
            if (aluon == 1'b0) begin
                if (i < 22) begin
                    i++;
                    #20 $display("%b %b %b %t", clkout, alucount, aluout, $time);
                end else begin
                    $finish;
                end
            end
        end

    endmodule
\end{verbatim}
